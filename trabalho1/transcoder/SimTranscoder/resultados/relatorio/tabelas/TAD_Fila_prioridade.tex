\centering
\caption{TAD Fila de Prioridade}
\label{table:1}
\begin{tabular}{|l|l|}
\hline
\textbf{construir}     & cria e inicializa uma instância da fila                            \\ \hline
\textbf{destruir}      & libera espaço alocado para uma instância da fila                   \\ \hline
\textbf{dado}          & define a estrutura de dados que manipulará a fila                  \\ \hline
\textbf{enfileirar(x)} & insere o elemento x na fila (de acordo com a sua prioridade)        \\ \hline
\textbf{desenfileirar} & remove o elemento de maior prioridade (primeiro ou último) da fila \\ \hline
\textbf{vazia}         & retorna verdadeiro se a fila não possuir elementos                 \\ \hline
\end{tabular}
\end{table}
